646         return defaultTemplate.build();
647      }

649      if (options == null)
650         options = optionsProvider.get();
651      logger.debug(">> searching params(%s)", this);
652      Set<? extends Image> images = getImages();
653      checkState(images.size() > 0, "no images present!");
654      Set<? extends Hardware> hardwaresToSearch = hardwares.get();
655      checkState(hardwaresToSearch.size() > 0, "no hardware profiles present!");
656

658      if (imageId != null) {

661            this.location = image.getLocation();

665      if (hardwareId != null) {

667         if (currentLocationWiderThan(hardware.getLocation()))
668            this.location = hardware.getLocation();
669      }

671      // if the user hasn't specified a location id, or an image or hardware
672      // with location, let's search scoped to the implicit one
673      if (location == null)
674         location = defaultLocation.get();
675

677         Iterable<? extends Image> supportedImages = findSupportedImages(images);
678         if (hardware == null)
679            hardware = resolveHardware(hardwaresToSearch, supportedImages);
680         image = resolveImage(hardware, supportedImages);
681      } else {
682         if (hardware == null)
683            hardware = resolveHardware(hardwaresToSearch, ImmutableSet.of(image));
684      }
685 
686      logger.debug("<<   matched image(%s) hardware(%s) location(%s)", image.getId(), hardware.getId(),
687            location.getId());

689   }

738               new Function<Hardware, Predicate<Image>>() {

741                  public Predicate<Image> apply(Hardware input) {
742                     return input.supportsImage();

744 
745               });

748               .getOnlyElement(supportsImagePredicates) : Predicates.<Image>or(supportsImagePredicates);

756      Predicate<Hardware> hardwarePredicate = buildHardwarePredicate();
757      Hardware hardware;
758      try {
759         hardware = hardwareOrdering.max(filter(hardwareCompatibleWithOurImages, hardwarePredicate));
760      } catch (NoSuchElementException exception) {
761         String message = format("no hardware profiles match params: %s", hardwarePredicate);
762         throw throwNoSuchElementExceptionAfterLoggingHardwareIds(message, hardwareCompatibleWithOurImages);
763      }

795         }

814   /**
815    * Like Ordering, but handle the case where there are multiple valid maximums
816    */
817   @SuppressWarnings("unchecked")
818   @VisibleForTesting
819   static <T, E extends T> List<E> multiMax(Comparator<T> ordering, Iterable<E> iterable) {
820      Iterator<E> iterator = iterable.iterator();
821      List<E> maxes = newArrayList(iterator.next());
822      E maxSoFar = maxes.get(0);
823      while (iterator.hasNext()) {
824         E current = iterator.next();
825         int comparison = ordering.compare(maxSoFar, current);
826         if (comparison == 0) {
827            maxes.add(current);
828         } else if (comparison < 0) {
829            maxes = newArrayList(current);
830            maxSoFar = current;
831         }
832      }
833      return maxes;
834   }
